-- Quick Installation Guide for lsrp_policevehicleparking
--
-- STEP 1: Database Setup
-- Execute this SQL in your database (HeidiSQL, phpMyAdmin, etc.):

CREATE TABLE IF NOT EXISTS `emergency_owned_vehicles` (
    `id` int(11) NOT NULL AUTO_INCREMENT,
    `license` varchar(60) NOT NULL,
    `vehicle_model` varchar(50) NOT NULL,
    `vehicle_plate` varchar(20) NOT NULL,
    `parking_zone` varchar(100) DEFAULT NULL,
    `vehicle_props` longtext NOT NULL,
    `status` varchar(20) NOT NULL DEFAULT 'parked',
    `purchased_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `stored_at` timestamp NULL DEFAULT CURRENT_TIMESTAMP,
    `last_retrieved_at` timestamp NULL DEFAULT NULL,
    PRIMARY KEY (`id`),
    UNIQUE KEY `vehicle_plate_unique` (`vehicle_plate`),
    KEY `license` (`license`),
    KEY `status` (`status`),
    KEY `parking_zone` (`parking_zone`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- The resource will automatically migrate legacy parked_vehicles rows into
-- emergency_owned_vehicles on startup.

-- STEP 2: Server Configuration
-- Add to your server.cfg (if not already present):
--
-- ensure oxmysql
-- ensure polyzone
-- ensure lsrp_policevehicleparking

-- STEP 3: Start the Resource
-- In your server console, run:
-- ensure lsrp_policevehicleparking
--
-- Or restart your server

-- STEP 4: Test In-Game
-- 1. Join the server
-- 2. Go to one of the parking zones:
--    - Legion Square: X: 215.9, Y: -809.1, Z: 30.7
--    - Airport: X: -796.9, Y: -2024.5, Z: 9.2
--    - Downtown: X: -340.5, Y: -874.5, Z: 31.1
-- 3. Press E to open the parking menu
-- 4. Get in a vehicle and click "Park Current Vehicle"
-- 5. Click "Retrieve Vehicle" to get it back

-- STEP 5: Customize (Optional)

-- - Add more parking zones
-- - Change interaction key
-- - Modify zone sizes
-- - Add parking fees
-- - Enable debug mode (Config.showParkingZoneDebug = true) to see zone boundaries

-- That's it! Your vehicle parking system is ready to use.
